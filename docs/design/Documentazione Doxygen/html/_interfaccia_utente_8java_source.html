488 private FileChooser creaFileChooser(String titolo) {
489 FileChooser fileChooser = new FileChooser();
490 fileChooser.setTitle(titolo);
491 fileChooser.getExtensionFilters().add(new FileChooser.ExtensionFilter("Archivio torneo (*.torneo)", "*.torneo"));
492
493 if (archivioCorrente != null) {
494 Path absolutePath = archivioCorrente.toAbsolutePath();
495 Path parent = absolutePath.getParent();
496 if (parent != null && Files.isDirectory(parent)) {
497 fileChooser.setInitialDirectory(parent.toFile());
498 }
499 fileChooser.setInitialFileName(absolutePath.getFileName().toString());
500 } else {
501 fileChooser.setInitialFileName("torneo.torneo");
502 }
