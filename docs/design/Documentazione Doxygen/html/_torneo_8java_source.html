255 private void validaSquadre(List<Squadra> nuoveSquadre) {
256 Set<String> nomiSquadre = new HashSet<>();
257
258 for (Squadra squadra : nuoveSquadre) {
259 if (squadra == null) {
260 throw new IllegalArgumentException("La lista delle squadre contiene elementi null.");
261 }
262 if (!squadra.isCompleta()) {
263 throw new IllegalArgumentException("Ogni squadra deve contenere esattamente 5 giocatori.");
264 }
265
266 String nomeNormalizzato = squadra.getNome().trim().toLowerCase(Locale.ROOT);
267 if (!nomiSquadre.add(nomeNormalizzato)) {
268 throw new IllegalArgumentException("Le squadre devono avere nomi univoci.");
